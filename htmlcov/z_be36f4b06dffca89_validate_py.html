15def infer_split(relative_path: Path) -> str: 

16 """ 

17 Infer the training split from a Path object. 

18 

19 The training split is inferred from the base part 

20 of a file path, where the filepath should be a relative path, 

21 with respect to its parent data directory. If the inferred split 

22 is not one of train, validation or test, an error is thrown. 

23 

24 Parameters 

25 ---------- 

26 relative_path : Path 

27 File path, relative to its data directory. 

28 

29 Returns 

30 ------- 

31 split : str 

32 Returns one of train, validation or test. 

33 """ 

34 for split in ["train", "validation", "test"]: 

35 if relative_path.parts[0].startswith(split): 

36 return split 

37 

38 # raise error if function does not return 

39 raise ValueError(f"{relative_path} should be in one of train, validation or test")

42def infer_audio_properties(audio_bytes: bytes) -> Dict[str, Union[int, float]]: 

43 """ 

44 Infer clip duration and sample rate from audio bytes. 

45 

46 Parameters 

47 ---------- 

48 audio_bytes : bytes 

49 Bytes like object containing audio to be read 

50 

51 Returns: 

52 -------- 

53 props: Dict[str, Union[int, float]] 

54 Dictionary containing the audio `sample_rate` as an int and 

55 audio `clip_duration` as a float. 

56 """ 

57 data, rate = soundfile.read(io.BytesIO(audio_bytes)) 

58 duration = len(data) / rate * 1e3 # in ms 

59 props = {"clip_duration": duration, "sample_rate": int(rate)} 

60 return props

63def audio_map_fn(file_path: Path) -> Dict[str, Union[int, float]]: 

64 """Map function for multiprocessing audio files.""" 

65 return infer_audio_properties(file_path.read_bytes())

68def validate_audio( 

69 audio_paths: List[Path], dataset_type: Optional[str] = "kws" 

70) -> None: 

71 """ 

72 Validate a portion of the audio files in the full dataset. 

73 

74 Validation will warn users if clip durations are inconsistent, as well as raise 

75 an error if audio sample rates differ between clips in the dataset. 

76 

77 Parameters 

78 ---------- 

79 audio_paths : List[Path] 

80 A list of path objects pointing to audio files in the dataset. 

81 dataset_type : str 

82 One of `kws` or `asr`, to toggle relevant warnings/errors. 

83 """ 

84 clip_durations = [] 

85 sample_rates = [] 

86 

87 with progress.Progress() as p: 

88 task_id = p.add_task( 

89 f"Reading {len(audio_paths)} audio samples", total=len(audio_paths) 

90 ) 

91 with multiprocessing.Pool(processes=multiprocessing.cpu_count()) as pool: 

92 for result in pool.imap_unordered(audio_map_fn, audio_paths): 

93 clip_durations.append(result["clip_duration"]) 

94 sample_rates.append(result["sample_rate"]) 

95 p.advance(task_id) 

96 

97 if dataset_type == "kws" and clip_durations.count(clip_durations[0]) != len( 

98 clip_durations 

99 ): 

100 # this isn't necessarily an error, clips will just be truncated/padded 

101 # to the desired length 

102 warnings.warn( 

103 "Clip durations are inconsistent. They will be cropped " 

104 "or padded at runtime." 

105 ) 

106 if sample_rates.count(sample_rates[0]) != len(sample_rates): 

107 raise ValueError( 

108 f"Found sample rates in the range ({min(sample_rates)}, " 

109 f"{max(sample_rates)}) Hz. Audio files require consistent sample " 

110 "rates before uploading." 

111 ) 

112 

113 print(f"Successfully validated {len(audio_paths)} audio files.")

116def validate_transcriptions( 

117 transcription_lines: List[str], 

118 audio_paths: List[Path], 

119) -> None: 

120 """ 

121 Validate transcriptions for ASR datasets. 

122 

123 Each line in a transcription file should have the format: 

124 `<filename>.<audio_format> <sentence>` 

125 where the first column is the filename in the same directory 

126 as the transcription file and the sentence portion is a space delimited 

127 string of words corresponding to the audio file. 

128 

129 Validation will throw an error if transcription entries do not contain 

130 filepaths that point to existing audio files in the dataset. 

131 

132 Parameters 

133 ---------- 

134 transcription_lines : List[str] 

135 A list of strings corresponding to the lines in each of the 

136 transcription files, with each line prepended with the absolute file path 

137 to the corresponding transcription file. 

138 audio_paths : List[Path] 

139 A list of path objects pointing to audio files in the dataset. 

140 """ 

141 expected_set = set() 

142 audio_set = set(audio_paths) 

143 for line in progress.track( 

144 transcription_lines, 

145 description=f"Reading {len(transcription_lines)} transcription entries", 

146 ): 

147 expected_set.add(Path(line.split()[0])) 

148 

149 if not expected_set.issubset(audio_set): 

150 missing = expected_set - audio_set 

151 raise FileNotFoundError( 

152 f"Found reference to {missing.pop()} in transcription, " 

153 "but file does not exist." 

154 ) 

155 

156 print(f"Successfully validated {len(transcription_lines)} transcription entries.")

159def validate_kws_formatting(audio_paths: List[Path]) -> Dict[str, List[str]]: 

160 """ 

161 Validate KWS datasets for proper directory structure and audio consistency. 

162 

163 Parameters 

164 ---------- 

165 audio_paths : List[Path] 

166 A list of path objects pointing to all audio files inside the data directory. 

167 

168 Returns 

169 ------- 

170 dataset_properties : Dict[str, List[str]] 

171 Dictionary containing a list of keywords that are inferred from 

172 the dataset. 

173 """ 

174 

175 # check validation failure conditions 

176 if len(audio_paths) == 0: 

177 raise FileNotFoundError( 

178 "Found no compatible audio files for kws datasets. " 

179 "Ensure audio is one of mp3, wav or flac." 

180 ) 

181 

182 all_splits = set() 

183 all_keywords = set() 

184 common_path = os.path.commonpath(audio_paths) 

185 for fpath in progress.track( 

186 audio_paths, description="Validating dataset formatting" 

187 ): 

188 rel_path = fpath.relative_to(common_path) if len(common_path) > 1 else fpath 

189 if rel_path.parts[0].startswith(("train", "validation", "test")): 

190 all_splits.add(infer_split(rel_path)) 

191 if not rel_path.parts[1].startswith("_"): 

192 all_keywords.add(rel_path.parts[1]) 

193 else: 

194 all_splits.add(rel_path.parts[0]) 

195 

196 # check split conditions are met 

197 if not set(("test", "train", "validation")).issubset(all_splits): 

198 raise FileNotFoundError("Did not find train/validation/test folders in dataset") 

199 

200 # make sure there are directories in the dataset archive with 

201 # names corresponding to viable keywords 

202 if len(all_keywords) == 0: 

203 raise FileNotFoundError("Did not find any keyword directories.") 

204 

205 # warn if no background noise directory 

206 if "background_noise" not in all_splits: 

207 warnings.warn( 

208 "No background_noise directory found. " 

209 "Consider adding for optimal kws performance in NengoEdge." 

210 ) 

211 

212 # warn about unused directories 

213 for split in all_splits: 

214 if split not in ["train", "validation", "test", "background_noise"]: 

215 warnings.warn( 

216 f"Found directory {split}, which will be unused by NengoEdge." 

217 ) 

218 

219 dataset_properties = {"allKeywords": sorted(list(all_keywords))} 

220 return dataset_properties

223def validate_asr_formatting( 

224 audio_paths: List[Path], transcription_paths: List[Path] 

225) -> None: 

226 """ 

227 Validate ASR datasets for proper directory structure, audio consistency, and 

228 transcription formatting. 

229 

230 Parameters 

231 ---------- 

232 audio_paths : List[Path] 

233 A list of path objects pointing to all audio 

234 files inside the data directory. 

235 transcription_paths : List[Path] 

236 A list of path objects pointing to all transcription 

237 files inside the data directory. 

238 """ 

239 

240 if len(audio_paths) == 0: 

241 raise FileNotFoundError( 

242 "Found no compatible audio files for asr datasets. " 

243 "Ensure audio is one of mp3, wav or flac." 

244 ) 

245 

246 if len(transcription_paths) == 0: 

247 raise FileNotFoundError( 

248 "Found no compatible transcription files for asr datasets. " 

249 "Ensure transcriptions are saved in txt files." 

250 ) 

251 

252 all_splits = set() 

253 file_paths = audio_paths + transcription_paths 

254 common_path = os.path.commonpath(file_paths) 

255 for fpath in progress.track( 

256 file_paths, description="Validating dataset formatting" 

257 ): 

258 all_splits.add( 

259 infer_split( 

260 fpath.relative_to(common_path) if len(common_path) > 1 else fpath 

261 ) 

262 ) 

263 

264 # check split conditions are met 

265 for split in ["train", "validation", "test"]: 

266 if split not in all_splits: 

267 raise FileNotFoundError( 

268 "Did not find train/validation/test folders in dataset" 

269 )

272def validate_dataset( 

273 dataset_type: str, data_directory: Path, validation_samples: Optional[int] 

274) -> List[Path]: 

275 """ 

276 Validate a dataset based on a given type and directory for that dataset. 

277 

278 Parameters 

279 ---------- 

280 dataset_type : str 

281 Type of dataset. One of kws or asr. 

282 data_directory : Path 

283 The directory in which the dataset resides. 

284 validation_samples : Optional[int] 

285 Maximum number of audio files to validate before packaging the dataset. 

286 """ 

287 # ensure we are only inspecting file paths, not directories 

288 file_paths = [fpath for fpath in data_directory.rglob("*") if fpath.is_file()] 

289 

290 if len(file_paths) == 0: 

291 raise FileNotFoundError(f"Found empty directory {data_directory}") 

292 

293 audio_paths = [f for f in file_paths if f.suffix in [".mp3", ".wav", ".flac"]] 

294 if validation_samples is None: 

295 validation_samples = len(audio_paths) 

296 if len(audio_paths) > 1e5: 

297 warnings.warn( 

298 f"Validating {len(audio_paths)} audio files may take many hours. " 

299 f"Run with --validation-steps 100000 to choose a random subset." 

300 ) 

301 

302 assert isinstance(validation_samples, int) 

303 audio_subset = random.sample(audio_paths, validation_samples) 

304 

305 if dataset_type == "kws": 

306 # validate formatting over all paths 

307 validate_kws_formatting(audio_paths) 

308 

309 # validate audio over subset if necessary 

310 validate_audio(audio_subset, dataset_type="kws") 

311 

312 elif dataset_type == "asr": 

313 txt_paths = [f for f in file_paths if f.suffix in [".txt"]] 

314 validate_asr_formatting(audio_paths, txt_paths) 

315 validate_audio(audio_subset, dataset_type="asr") 

316 

317 # add absolute file path to transcription entries in order 

318 # to search for corresponding audio files 

319 transcription_lines = [ 

320 f"{f.parent}/{line}" 

321 for f in txt_paths 

322 for line in f.read_text().splitlines() 

323 ] 

324 validate_transcriptions(transcription_lines, audio_paths) 

325 

326 return file_paths
